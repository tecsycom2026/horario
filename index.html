⚠️ Primer uso: crear tablas en Supabase (clic para ver instrucciones) Ve a tu proyecto Supabase → SQL Editor → pega y ejecuta este SQL: 
-- Tabla de personas
CREATE TABLE IF NOT EXISTS persons (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text NOT NULL,
  sort_order integer NOT NULL DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

-- Tabla de configuración
CREATE TABLE IF NOT EXISTS schedule_config (
  id integer PRIMARY KEY DEFAULT 1,
  ref_monday date NOT NULL DEFAULT '2025-01-06',
  updated_at timestamptz DEFAULT now()
);

-- Insertar config inicial
INSERT INTO schedule_config (id, ref_monday)
VALUES (1, '2025-01-06')
ON CONFLICT (id) DO NOTHING;

-- Permisos públicos (anon)
ALTER TABLE persons ENABLE ROW LEVEL SECURITY;
ALTER TABLE schedule_config ENABLE ROW LEVEL SECURITY;

CREATE POLICY "public read persons" ON persons FOR SELECT USING (true);
CREATE POLICY "public write persons" ON persons FOR ALL USING (true);
CREATE POLICY "public read config" ON schedule_config FOR SELECT USING (true);
CREATE POLICY "public write config" ON schedule_config FOR ALL USING (true);
